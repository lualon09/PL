//Calcula el máximo de un array

calculaMax(arr:Int[], length:Int) -> Int{
    Int max = arr[0];

    for(Int i = 0; i < length; i = i + 1){
        if(arr[i] > max){
            max = arr[i];
        }
    }

    show(max);                  //mostramos el máximo

    return max;
}
